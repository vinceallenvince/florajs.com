  2 /**
  3  * Creates a new Agent.
  4  *
  5  * Agents are basic Flora elements that respond to forces like gravity, attraction,
  6  * repulsion, etc. They can also chase after other Agents, organize with other Agents
  7  * in a flocking behavior, and steer away from obstacles. They can also follow the mouse.
  8  *
  9  * @param {Object} [opt_options=] A map of initial properties.
 10  * @constructor
 11  * @extends Mover
 12  */
 13 function Agent(opt_options) {
 14   var options = opt_options || {};
 15   options.name = options.name || 'Agent';
 16   exports.Mover.call(this, options);
 17 }
 18 exports.Utils.extend(Agent, exports.Mover);
 19 
 20 /**
 21  * Initializes an instance.
 22  *
 23  * @param {Object} [opt_options=] A map of initial properties.
 24  * @param {boolean} [opt_options.followMouse = false] If true, object will follow mouse.
 25  * @param {number} [opt_options.maxSteeringForce = 10] Set the maximum strength of any steering force.
 26  * @param {Object} [opt_options.seekTarget = null] An object to seek.
 27  * @param {boolean} [opt_options.flocking = false] Set to true to apply flocking forces to this object.
 28  * @param {number} [opt_options.desiredSeparation = Twice the object's default width] Sets the desired separation from other objects when flocking = true.
 29  * @param {number} [opt_options.separateStrength = 1] The strength of the force to apply to separating when flocking = true.
 30  * @param {number} [opt_options.alignStrength = 1] The strength of the force to apply to aligning when flocking = true.
 31  * @param {number} [opt_options.cohesionStrength = 1] The strength of the force to apply to cohesion when flocking = true.
 32  * @param {Object} [opt_options.flowField = null] If a flow field is set, object will use it to apply a force.
 33  * @param {Array} [opt_options.sensors = []] A list of sensors attached to this object.
 34  * @param {Array} [opt_options.color = [197, 177, 115]] Color.
 35  * @param {number} [opt_options.borderWidth = 0] Border width.
 36  * @param {string} [opt_options.borderStyle = 'none'] Border style.
 37  * @param {string|Array} [opt_options.borderColor = 'transparent'] Border color.
 38  * @param {number} [opt_options.borderRadius = 0] Border radius.
 39  */
 40 Agent.prototype.init = function(opt_options) {
 41 
 42   var options = opt_options || {};
 43   Agent._superClass.prototype.init.call(this, options);
 44 
 45   this.followMouse = !!options.followMouse;
 46   this.maxSteeringForce = options.maxSteeringForce || 10;
 47   this.seekTarget = options.seekTarget || null;
 48   this.flocking = !!options.flocking;
 49   this.desiredSeparation = options.desiredSeparation === 0 ? 0 : options.desiredSeparation || this.width * 2;
 50   this.separateStrength = options.separateStrength === 0 ? 0 : options.separateStrength || 0.3;
 51   this.alignStrength = options.alignStrength === 0 ? 0 : options.alignStrength || 0.2;
 52   this.cohesionStrength = options.cohesionStrength === 0 ? 0 : options.cohesionStrength || 0.1;
 53   this.flowField = options.flowField || null;
 54   this.sensors = options.sensors || [];
 55 
 56   this.color = options.color || [197, 177, 115];
 57   this.borderWidth = options.borderWidth || 0;
 58   this.borderStyle = options.borderStyle || 'none';
 59   this.borderColor = options.borderColor || 'transparent';
 60   this.borderRadius = options.borderRadius || this.sensors.length ? 100 : 0;
 61 
 62   //
 63 
 64   this.separateSumForceVector = new Burner.Vector(); // used in Agent.separate()
 65   this.alignSumForceVector = new Burner.Vector(); // used in Agent.align()
 66   this.cohesionSumForceVector = new Burner.Vector(); // used in Agent.cohesion()
 67   this.followTargetVector = new Burner.Vector(); // used in Agent.applyForces()
 68   this.followDesiredVelocity = new Burner.Vector(); // used in Agent.follow()
 69 };
 70 
 71 Agent.prototype.name = 'Agent';

 73 /**
 74  * Applies Agent-specific forces.
 75  *
 76  * @returns {Object} This object's acceleration vector.
 77  */
 78 Agent.prototype.applyForces = function() {
 79 
 80   var i, max, sensorActivated, dir, sensor, r, theta, x, y,
 81       liquids = Burner.System._caches.Liquid,
 82       attractors = Burner.System._caches.Attractor,
 83       repellers = Burner.System._caches.Repeller,
 84       heat = Burner.System._caches.Heat;
 85 
 86   if (liquids && liquids.list.length > 0) { // liquid
 87     for (i = 0, max = liquids.list.length; i < max; i += 1) {
 88       if (this.id !== liquids.list[i].id && exports.Utils.isInside(this, liquids.list[i])) {
 89         this.applyForce(this.drag(liquids.list[i]));
 90       }
 91     }
 92   }
 93 
 94   if (attractors && attractors.list.length > 0) { // attractor
 95     for (i = 0, max = attractors.list.length; i < max; i += 1) {
 96       if (this.id !== attractors.list[i].id) {
 97         this.applyForce(this.attract(attractors.list[i]));
 98       }
 99     }
100   }
101 
102   if (repellers && repellers.list.length > 0) { // repeller
103     for (i = 0, max = repellers.list.length; i < max; i += 1) {
104       if (this.id !== repellers.list[i].id) {
105         this.applyForce(this.attract(repellers.list[i]));
106       }
107     }
108   }
109 
110   if (this.sensors.length > 0) { // Sensors
111     for (i = 0, max = this.sensors.length; i < max; i += 1) {
112 
113       sensor = this.sensors[i];
114 
115       r = sensor.offsetDistance; // use angle to calculate x, y
116       theta = exports.Utils.degreesToRadians(this.angle + sensor.offsetAngle);
117       x = r * Math.cos(theta);
118       y = r * Math.sin(theta);
119 
120       sensor.location.x = this.location.x;
121       sensor.location.y = this.location.y;
122       sensor.location.add(new Burner.Vector(x, y)); // position the sensor
123 
124       if (i) {
125         sensor.borderStyle = 'none';
126       }
127 
128       if (sensor.activated) {
129         this.applyForce(sensor.getActivationForce(this));
130         sensorActivated = true;
131       }
132 
133     }
134   }
135 
136   /**
137    * If no sensors were activated and this.motorSpeed != 0,
138    * apply a force in the direction of the current velocity.
139    */
140   if (!sensorActivated && this.motorSpeed) {
141     dir = exports.Utils.clone(this.velocity);
142     dir.normalize();
143     if (this.velocity.mag() > this.motorSpeed) { // decelerate to defaultSpeed
144       dir.mult(-this.motorSpeed);
145     } else {
146       dir.mult(this.motorSpeed);
147     }
148     this.applyForce(dir); // constantly applies a force
149   }
150 
151   if (this.followMouse) { // follow mouse
152     var t = {
153       location: new Burner.Vector(Burner.System.mouse.location.x,
154           Burner.System.mouse.location.y)
155     };
156     this.applyForce(this._seek(t));
157   }
158 
159   if (this.seekTarget) { // seek target
160     this.applyForce(this._seek(this.seekTarget));
161   }
162 
163   if (this.flowField) { // follow flow field
164     var res = this.flowField.resolution,
165       col = Math.floor(this.location.x/res),
166       row = Math.floor(this.location.y/res),
167       loc, target;
168 
169     if (this.flowField.field[col]) {
170       loc = this.flowField.field[col][row];
171       if (loc) { // sometimes loc is not available for edge cases
172         this.followTargetVector.x = loc.x;
173         this.followTargetVector.y = loc.y;
174       } else {
175         this.followTargetVector.x = this.location.x;
176         this.followTargetVector.y = this.location.y;
177       }
178       target = {
179         location: this.followTargetVector
180       };
181       this.applyForce(this.follow(target));
182     }
183 
184   }
185 
186   if (this.flocking) {
187     this.flock(Burner.System.getAllItemsByName('Agent'));
188   }
189 
190   return this.acceleration;
191 };

193 /**
194  * Calculates a steering force to apply to an object following another object.
195  * Agents with flow fields will use this method to calculate a steering force.
196  *
197  * @param {Object} target The object to follow.
198  * @returns {Object} The force to apply.
199  */
200 Agent.prototype.follow = function(target) {
201 
202   this.followDesiredVelocity.x = target.location.x;
203   this.followDesiredVelocity.y = target.location.y;
204 
205   this.followDesiredVelocity.mult(this.maxSpeed);
206   this.followDesiredVelocity.sub(this.velocity);
207   this.followDesiredVelocity.limit(this.maxSteeringForce);
208 
209   return this.followDesiredVelocity;
210 };
211 
212 /**
213  * Bundles flocking behaviors (separate, align, cohesion) into one call.
214  *
215  * @returns {Object} This object's acceleration vector.
216  */
217 Agent.prototype.flock = function(elements) {
218 
219   this.applyForce(this.separate(elements).mult(this.separateStrength));
220   this.applyForce(this.align(elements).mult(this.alignStrength));
221   this.applyForce(this.cohesion(elements).mult(this.cohesionStrength));
222   return this.acceleration;
223 };

225 /**
226  * Loops through a passed elements array and calculates a force to apply
227  * to avoid all elements.
228  *
229  * @param {array} elements An array of Flora elements.
230  * @returns {Object} A force to apply.
231  */
232 Agent.prototype.separate = function(elements) {
233 
234   var i, max, element, diff, d,
235   sum, count = 0, steer;
236 
237   this.separateSumForceVector.x = 0;
238   this.separateSumForceVector.y = 0;
239   sum = this.separateSumForceVector;
240 
241   for (i = 0, max = elements.length; i < max; i += 1) {
242     element = elements[i];
243     if (this.className === element.className && this.id !== element.id) {
244 
245       d = this.location.distance(element.location);
246 
247       if ((d > 0) && (d < this.desiredSeparation)) {
248         diff = Burner.Vector.VectorSub(this.location, element.location);
249         diff.normalize();
250         diff.div(d);
251         sum.add(diff);
252         count += 1;
253       }
254     }
255   }
256   if (count > 0) {
257     sum.div(count);
258     sum.normalize();
259     sum.mult(this.maxSpeed);
260     sum.sub(this.velocity);
261     sum.limit(this.maxSteeringForce);
262     return sum;
263   }
264   return new Burner.Vector();
265 };

267 /**
268  * Loops through a passed elements array and calculates a force to apply
269  * to align with all elements.
270  *
271  * @param {array} elements An array of Flora elements.
272  * @returns {Object} A force to apply.
273  */
274 Agent.prototype.align = function(elements) {
275 
276   var i, max, element, d,
277     neighbordist = this.width * 2,
278     sum, count = 0, steer;
279 
280   this.alignSumForceVector.x = 0;
281   this.alignSumForceVector.y = 0;
282   sum = this.alignSumForceVector;
283 
284   for (i = 0, max = elements.length; i < max; i += 1) {
285     element = elements[i];
286     d = this.location.distance(element.location);
287 
288     if ((d > 0) && (d < neighbordist)) {
289       if (this.className === element.className && this.id !== element.id) {
290         sum.add(element.velocity);
291         count += 1;
292       }
293     }
294   }
295 
296   if (count > 0) {
297     sum.div(count);
298     sum.normalize();
299     sum.mult(this.maxSpeed);
300     sum.sub(this.velocity);
301     sum.limit(this.maxSteeringForce);
302     return sum;
303   }
304   return new Burner.Vector();
305 };

307 /**
308  * Loops through a passed elements array and calculates a force to apply
309  * to stay close to all elements.
310  *
311  * @param {array} elements An array of Flora elements.
312  * @returns {Object} A force to apply.
313  */
314 Agent.prototype.cohesion = function(elements) {
315 
316   var i, max, element, d,
317     neighbordist = 10,
318     sum, count = 0, desiredVelocity, steer;
319 
320   this.cohesionSumForceVector.x = 0;
321   this.cohesionSumForceVector.y = 0;
322   sum = this.cohesionSumForceVector;
323 
324   for (i = 0, max = elements.length; i < max; i += 1) {
325     element = elements[i];
326     d = this.location.distance(element.location);
327 
328     if ((d > 0) && (d < neighbordist)) {
329       if (this.className === element.className && this.id !== element.id) {
330         sum.add(element.location);
331         count += 1;
332       }
333     }
334   }
335 
336   if (count > 0) {
337     sum.div(count);
338     sum.sub(this.location);
339     sum.normalize();
340     sum.mult(this.maxSpeed);
341     sum.sub(this.velocity);
342     sum.limit(this.maxSteeringForce);
343     return sum;
344   }
345   return new Burner.Vector();
346 };

348 /**
349  * Returns this object's location.
350  *
351  * @param {string} [type] If no type is supplied, returns a clone of this object's location.
352                           Accepts 'x', 'y' to return their respective values.
353  * @returns {boolean} Returns true if the object is outside the world.
354  */
355 Agent.prototype.getLocation = function (type) {
356 
357   if (!type) {
358     return new Burner.Vector(this.location.x, this.location.y);
359   } else if (type === 'x') {
360     return this.location.x;
361   } else if (type === 'y') {
362     return this.location.y;
363   }
364 };

374

376     return new Burner.Vector(this.location.x, this.location.y);

378     return this.velocity.x;
379   } else if (type === 'y') {
380     return this.velocity.y;
