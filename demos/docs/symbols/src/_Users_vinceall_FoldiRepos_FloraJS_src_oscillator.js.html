  2 /**
  3  * Creates a new Oscillator.
  4  *
  5  * Oscillators simulate wave patterns and move according to
  6  * amplitude and angular velocity. Oscillators are not affected
  7  * by gravity or friction.
  8  *
  9  * @constructor
 10  * @extends Burner.Item
 11  *
 12  * @param {Object} [opt_options=] A map of initial properties.
 13  */
 14 function Oscillator(opt_options) {
 15   var options = opt_options || {};
 16   options.name = options.name || 'Oscillator';
 17   Burner.Item.call(this, options);
 18 }
 19 exports.Utils.extend(Oscillator, Burner.Item);
 20 
 21 /**
 22  * Initializes an instance.
 23  *
 24  * @param {Object} [opt_options=] A map of initial properties.
 25  * @param {Object} [opt_options.initialLocation = The center of the world] The object's initial location.
 26  * @param {Object} [opt_options.lastLocation = {x: 0, y: 0}] The object's last location. Used to calculate
 27  *    angle if pointToDirection = true.
 28  * @param {Object} [opt_options.amplitude = {x: world width, y: world height}] Sets amplitude, the distance from the object's
 29  *    initial location (center of the motion) to either extreme.
 30  * @param {Object} [opt_options.acceleration = {x: 0, y: 0}] The object's acceleration. Oscillators have a
 31  *    constant acceleration.
 32  * @param {Object} [opt_options.aVelocity = new Vector()] Angular velocity.
 33  * @param {boolean} [opt_options.isStatic = false] If true, object will not move.
 34  * @param {boolean} [opt_options.isPerlin = true] If set to true, object will use Perlin Noise to calculate its location.
 35  * @param {number} [opt_options.perlinSpeed = 0.005] If isPerlin = true, perlinSpeed determines how fast the object location moves through the noise space.
 36  * @param {number} [opt_options.perlinTime = 0] Sets the Perlin Noise time.
 37  * @param {number} [opt_options.perlinAccelLow = -2] The lower bound of acceleration when isPerlin = true.
 38  * @param {number} [opt_options.perlinAccelHigh = 2] The upper bound of acceleration when isPerlin = true.
 39  * @param {number} [opt_options.offsetX = Math.random() * 10000] The x offset in the Perlin Noise space.
 40  * @param {number} [opt_options.offsetY = Math.random() * 10000] The y offset in the Perlin Noise space.
 41  * @param {number} [opt_options.width = 20] Width.
 42  * @param {number} [opt_options.height = 20] Height.
 43  * @param {Array} [opt_options.color = [200, 100, 0]] Color.
 44  * @param {number} [opt_options.borderWidth = this.width / 4] Border width.
 45  * @param {string} [opt_options.borderStyle = 'solid'] Border style.
 46  * @param {Array} [opt_options.borderColor = [255, 150, 0]] Border color.
 47  * @param {number} [opt_options.borderRadius = 100] Border radius.
 48  * @param {number} [opt_options.boxShadowSpread = this.width / 4] Box-shadow spread.
 49  * @param {Array} [opt_options.boxShadowColor = [147, 199, 196]] Box-shadow color.
 50  */
 51 Oscillator.prototype.init = function(opt_options) {
 52 
 53   var options = opt_options || {};
 54 
 55   this.initialLocation = options.initialLocation ||
 56       new Burner.Vector(this.world.bounds[1] / 2, this.world.bounds[2] / 2);
 57   this.lastLocation = new Burner.Vector();
 58   this.amplitude = options.amplitude || new Burner.Vector(this.world.bounds[1] / 2 - this.width,
 59       this.world.bounds[2] / 2 - this.height);
 60   this.acceleration = options.acceleration || new Burner.Vector(0.01, 0);
 61   this.aVelocity = options.aVelocity || new Burner.Vector();
 62   this.isStatic = !!options.isStatic;
 63 
 64   this.isPerlin = !!options.isPerlin;
 65   this.perlinSpeed = options.perlinSpeed || 0.005;
 66   this.perlinTime = options.perlinTime || 0;
 67   this.perlinAccelLow = options.perlinAccelLow || -2;
 68   this.perlinAccelHigh = options.perlinAccelHigh || 2;
 69   this.perlinOffsetX = options.perlinOffsetX || Math.random() * 10000;
 70   this.perlinOffsetY = options.perlinOffsetY || Math.random() * 10000;
 71 
 72   this.width = options.width || 20;
 73   this.height = options.height || 20;
 74   this.color = options.color || [200, 100, 0];
 75   this.borderWidth = options.borderWidth || 0;
 76   this.borderStyle = options.borderStyle || 'solid';
 77   this.borderColor = options.borderColor || [255, 150, 50];
 78   this.borderRadius = options.borderRadius || 100;
 79   this.boxShadowSpread = options.boxShadowSpread || 0;
 80   this.boxShadowColor = options.boxShadowColor || [200, 100, 0];
 81 };

 83 Oscillator.prototype.name = 'Oscillator';

 85 /**
 86  * Updates the oscillator's properties.
 87  */
 88 Oscillator.prototype.step = function () {
 89 
 90   var world = this.world, velDiff;
 91 
 92   if (this.beforeStep) {
 93     this.beforeStep.apply(this);
 94   }
 95 
 96   if (!this.isStatic && !this.isPressed) {
 97 
 98     if (this.isPerlin) {
 99       this.perlinTime += this.perlinSpeed;
100       this.aVelocity.x =  exports.Utils.map(exports.SimplexNoise.noise(this.perlinTime + this.perlinOffsetX, 0, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
101       this.aVelocity.y =  exports.Utils.map(exports.SimplexNoise.noise(0, this.perlinTime + this.perlinOffsetY, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
102     } else {
103       this.aVelocity.add(this.acceleration); // add acceleration
104     }
105 
106     this.location.x = this.initialLocation.x + Math.sin(this.aVelocity.x) * this.amplitude.x;
107     this.location.y = this.initialLocation.y + Math.sin(this.aVelocity.y) * this.amplitude.y;
108 
109     if (this.pointToDirection) { // object rotates toward direction
110         velDiff = Burner.Vector.VectorSub(this.location, this.lastLocation);
111         this.angle = exports.Utils.radiansToDegrees(Math.atan2(velDiff.y, velDiff.x));
112     }
113 
114     if (this.controlCamera) {
115       this._checkCameraEdges();
116     }
117 
118     if (this.checkEdges || this.wrapEdges) {
119       this._checkWorldEdges(world);
120     }
121 
122     if (this.lifespan > 0) {
123       this.lifespan -= 1;
124     }
125 
126     if (this.afterStep) {
127       this.afterStep.apply(this);
128     }
129 
130     if (this.pointToDirection) {
131       this.lastLocation.x = this.location.x;
132       this.lastLocation.y = this.location.y;
133     }
134   }
135 };
136 exports.Oscillator = Oscillator;
