 17 /**
 18  * Initializes an instance.
 19  *
 20  * @param {Object} [opt_options=] A map of initial properties.
 21  * @param {number} [opt_options.width = 10] Width
 22  * @param {number} [opt_options.height = 10] Height
 23  * @param {string|Array} [opt_options.color = [255, 255, 255]] Color.
 24  * @param {number} [opt_options.motorSpeed = 2] Motor speed
 25  * @param {number} [opt_options.angle = 10] Angle
 26  * @param {boolean} [opt_options.pointToDirection = true] If true, object will point in the direction it's moving.
 27  * @param {boolean} [opt_options.draggable = false] If true, object can move via drag and drop.
 28  * @param {Object} [opt_options.parent = null] A parent object. If set, object will be fixed to the parent relative to an offset distance.
 29  * @param {number} [opt_options.offsetDistance = 30] The distance from the center of the object's parent.
 30  * @param {function} [opt_options.beforeStep = null] A function to run before the step() function.
 31  * @param {function} [opt_options.afterStep = null] A function to run after the step() function.
 32  */
 33 Mover.prototype.init = function(options) {
 34 
 35   this.width = options.width || 20;
 36   this.height = options.height || 20;
 37   this.color = options.color || [255, 255, 255];
 38   this.motorSpeed = options.motorSpeed || 0;
 39   this.angle = options.angle || 0;
 40   this.pointToDirection = options.pointToDirection === false ? false : true;
 41   this.draggable = !!options.draggable;
 42   this.parent = options.parent || null;
 43   this.offsetDistance = options.offsetDistance || 30;
 44   this.beforeStep = options.beforeStep || null;
 45   this.afterStep = options.afterStep || null;
 46 
 47   //
 48 
 49   this.isMouseOut = false;
 50   this.isPressed = false;
 51 
 52   var mouseover = (function (me) {
 53     return (function(e) {
 54       me.mouseover(e, me);
 55     });
 56   })(this);
 57 
 58   var mousedown = (function (me) {
 59     return (function(e) {
 60       me.mousedown(e, me);
 61     });
 62   })(this);
 63 
 64   var mousemove = (function (me) {
 65     return (function(e) {
 66       me.mousemove(e, me);
 67     });
 68   })(this);
 69 
 70   var mouseup = (function (me) {
 71     return (function(e) {
 72       me.mouseup(e, me);
 73     });
 74   })(this);
 75 
 76   var mouseout = (function (me) {
 77     return (function(e) {
 78       me.mouseout(e, me);
 79     });
 80   })(this);
 81 
 82   if (this.draggable) {
 83     exports.Utils.addEvent(this.el, 'mouseover', mouseover);
 84     exports.Utils.addEvent(this.el, 'mousedown', mousedown);
 85     exports.Utils.addEvent(this.el, 'mousemove', mousemove);
 86     exports.Utils.addEvent(this.el, 'mouseup', mouseup);
 87     exports.Utils.addEvent(this.el, 'mouseout', mouseout);
 88   }
 89 };

 98 Mover.prototype.mousedown = function(e) {
 99 
100   var touch, target = e.target || e.srcElement; // <= IE* uses srcElement
101 
102   if (e.changedTouches) {
103     touch = e.changedTouches[0];
104   }
105 
106   if (e.pageX && e.pageY) {
107     this.offsetX = e.pageX - target.offsetLeft;
108     this.offsetY = e.pageY - target.offsetTop;
109   } else if (e.clientX && e.clientY) {
110     this.offsetX = e.clientX - target.offsetLeft;
111     this.offsetY = e.clientY - target.offsetTop;
112   } else if (touch) {
113     this.offsetX = touch.pageX - target.offsetLeft;
114     this.offsetY = touch.pageY - target.offsetTop;
115   }
116 
117   this.isPressed = true;
118   this.isMouseOut = false;
119 };

121 /**
122  * Called by a mousemove event listener.
123  *
124  * @param {Object} e The event object passed by the listener.
125  */
126 Mover.prototype.mousemove = function(e) {
127 
128   var x, y, touch;
129 
130   if (this.isPressed) {
131 
132     this.isMouseOut = false;
133 
134     if (e.changedTouches) {
135       touch = e.changedTouches[0];
136     }
137 
138     if (e.pageX && e.pageY) {
139       x = e.pageX - this.world.el.offsetLeft;
140       y = e.pageY - this.world.el.offsetTop;
141     } else if (e.clientX && e.clientY) {
142       x = e.clientX - this.world.el.offsetLeft;
143       y = e.clientY - this.world.el.offsetTop;
144     } else if (touch) {
145       x = touch.pageX - this.world.el.offsetLeft;
146       y = touch.pageY - this.world.el.offsetTop;
147     }
148 
149     if (exports.Utils.mouseIsInsideWorld(this.world)) {
150       this.location = new Burner.Vector(x, y);
151     } else {
152       this.isPressed = false;
153     }
154   }
155 };

167  * Called by a mouseout event listener.

169  * @param {Object} e The event object passed by the listener.
170  * @param {Object} obj The object associated with the event target.
171  */
172 Mover.prototype.mouseout = function(e, obj) {
173 
174 
175 
176   var me = obj, mouse = Burner.System.mouse,
177       x, y;
178 
179   if (obj.isPressed) {
180 
181     obj.isMouseOut = true;
182 
183     obj.mouseOutInterval = setInterval(function () { // if mouse is too fast for block update, update via an interval until it catches up
184 
185       if (me.isPressed && me.isMouseOut) {
186 
187         x = mouse.location.x - me.world.el.offsetLeft;
188         y = mouse.location.y - me.world.el.offsetTop;
189 
190         me.location = new Burner.Vector(x, y);
191       }
192     }, 16);
193   }
194 };

196 /**
197  * Calculates location via sum of acceleration + velocity.
198  *
199  * @returns {number} The total number of times step has been executed.
200  */
201 Mover.prototype.step = function() {
202 
203   var friction, r, theta, x, y;
204 
205   if (this.beforeStep) {
206     this.beforeStep.apply(this);
207   }
208 
209   if (!this.isStatic && !this.isPressed) {
210 
211     // start apply forces
212 
213     if (this.world.c) { // friction
214       friction = exports.Utils.clone(this.velocity);
215       friction.mult(-1);
216       friction.normalize();
217       friction.mult(this.world.c);
218       this.applyForce(friction);
219     }
220     this.applyForce(this.world.gravity); // gravity
221 
222     if (this.applyForces) { // !! rename this
223       this.applyForces();
224     }
225 
226     // end apply forces
227 
228     this.velocity.add(this.acceleration); // add acceleration
229 
230     this.velocity.limit(this.maxSpeed, this.minSpeed);
231 
232     this.location.add(this.velocity); // add velocity
233     if (this.pointToDirection) { // object rotates toward direction
234       if (this.velocity.mag() > 0.1) {
235         this.angle = exports.Utils.radiansToDegrees(Math.atan2(this.velocity.y, this.velocity.x));
236       }
237     }
238   }
239 
240   if (this.controlCamera) { // check camera after velocity/location calculation
241     this._checkCameraEdges();
242   }
243 
244   if (this.checkWorldEdges) {
245     this._checkWorldEdges();
246   }
247 
248   if (this.parent) { // parenting
249 
250     if (this.offsetDistance) {
251 
252       r = this.offsetDistance; // use angle to calculate x, y
253       theta = exports.Utils.degreesToRadians(this.parent.angle + this.offsetAngle);
254       x = r * Math.cos(theta);
255       y = r * Math.sin(theta);
256 
257       this.location.x = this.parent.location.x;
258       this.location.y = this.parent.location.y;
259       this.location.add(new Burner.Vector(x, y)); // position the child
260 
261     } else {
262       this.location = this.parent.location;
263     }
264   }
265 
266   this.acceleration.mult(0);
267 
268   if (this.life < this.lifespan) {
269     this.life += 1;
270   } else if (this.lifespan !== -1) {
271     Burner.System.destroyItem(this);
272   }
273 
274   if (this.afterStep) {
275     this.afterStep.apply(this);
276   }
277 };

279 /**
280  * Calculates a steering force to apply to an object seeking another object.
281  *
282  * @param {Object} target The object to seek.
283  * @returns {Object} The force to apply.
284  * @private
285  */
286 Mover.prototype._seek = function(target) {
287 
288   var world = this.world,
289     desiredVelocity = Burner.Vector.VectorSub(target.location, this.location),
290     distanceToTarget = desiredVelocity.mag();
291 
292   desiredVelocity.normalize();
293 
294   if (distanceToTarget < world.bounds[1] / 2) {
295     var m = exports.Utils.map(distanceToTarget, 0, world.bounds[1] / 2, 0, this.maxSpeed);
296     desiredVelocity.mult(m);
297   } else {
298     desiredVelocity.mult(this.maxSpeed);
299   }
300 
301   desiredVelocity.sub(this.velocity);
302   desiredVelocity.limit(this.maxSteeringForce);
303 
304   return desiredVelocity;
305 };

307 /**
308  * Checks if object is within range of a world edge. If so, steers the object
309  * in the opposite direction.
310  * @private
311  */
312 Mover.prototype._checkAvoidEdges = function() {
313 
314   var maxSpeed, desiredVelocity;
315 
316   if (this.location.x < this.avoidWorldEdgesStrength) {
317     maxSpeed = this.maxSpeed;
318   } else if (this.location.x > this.world.bounds[1] - this.avoidWorldEdgesStrength) {
319     maxSpeed = -this.maxSpeed;
320   }
321   if (maxSpeed) {
322     desiredVelocity = new Burner.Vector(maxSpeed, this.velocity.y);
323     desiredVelocity.sub(this.velocity);
324     desiredVelocity.limit(this.maxSteeringForce);
325     this.applyForce(desiredVelocity);
326   }
327 
328   if (this.location.y < this.avoidWorldEdgesStrength) {
329     maxSpeed = this.maxSpeed;
330   } else if (this.location.y > this.world.bounds[2] - this.avoidWorldEdgesStrength) {
331     maxSpeed = -this.maxSpeed;
332   }
333   if (maxSpeed) {
334     desiredVelocity = new Burner.Vector(this.velocity.x, maxSpeed);
335     desiredVelocity.sub(this.velocity);
336     desiredVelocity.limit(this.maxSteeringForce);
337     this.applyForce(desiredVelocity);
338   }
339 };

341 /**
342  * Calculates a force to apply to simulate drag on an object.
343  *
344  * @param {Object} target The object that is applying the drag force.
345  * @returns {Object} A force to apply.
346  */
347 Mover.prototype.drag = function(target) {
348 
349 
350 
351   var speed = this.velocity.mag(),
352     dragMagnitude = -1 * target.c * speed * speed, // drag magnitude
353     drag = exports.Utils.clone(this.velocity);
354 
355   drag.normalize(); // drag direction
356   drag.mult(dragMagnitude);
357 
358   return drag;
359 };

361 /**
362  * Calculates a force to apply to simulate attraction on an object.
363  *
364  * @param {Object} attractor The attracting object.
365  * @returns {Object} A force to apply.
366  */
367 Mover.prototype.attract = function(attractor) {
368 
369   var force = Burner.Vector.VectorSub(attractor.location, this.location),
370     distance, strength;
371 
372   distance = force.mag();
373   distance = exports.Utils.constrain(distance, this.width * this.height/8, attractor.width * attractor.height); // min = scale/8 (totally arbitrary); max = scale; the size of the attractor
374   force.normalize();
375   strength = (attractor.G * attractor.mass * this.mass) / (distance * distance);
376   force.mult(strength);
377 
378   return force;
379 };

381 /**
382  * Determines if this object is inside another.
383  *
384  * @param {Object} container The containing object.
385  * @returns {boolean} Returns true if the object is inside the container.
386  */
387 Mover.prototype.isInside = function(container) {
388 
389   if (container) {
390     if (this.location.x + this.width / 2 > container.location.x - container.width / 2 &&
391       this.location.x - this.width / 2 < container.location.x + container.width / 2 &&
392       this.location.y + this.height / 2 > container.location.y - container.height / 2 &&
393       this.location.y - this.height / 2 < container.location.y + container.height / 2) {
394       return true;
395     }
396   }
397   return false;
398 };
