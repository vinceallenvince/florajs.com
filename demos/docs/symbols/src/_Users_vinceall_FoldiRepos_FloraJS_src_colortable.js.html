  2 /**
  3  * Creates a new ColorTable.
  4  *
  5  * Use a color table to create a map of keywords to color ranges.
  6  * Instead of manually passing start and end colors when creating
  7  * color palettes, you can use getColor() and pass a keyword to
  8  * receive the start and end colors.
  9  *
 10  * @example
 11  * var heat = defaultColors.getColor('heat');
 12  * console.log(heat.startColor); // -> [255, 132, 86]
 13  * console.log(heat.endColor); // -> [175, 47, 0]
 14  *
 15  * @constructor
 16  */
 17 function ColorTable() {
 18 }
 19 
 20 /**
 21  * Adds a key to the color table with start and end color values.
 22  *
 23  * @param {Object} options A set of required options
 24  *    that includes:
 25  *    options.name {number} The name of the entry in the color table.
 26  *    options.startColor {Array} The beginning color of the color range.
 27  *    options.endColor {Array} The end color of the color range.
 28  *
 29  * @returns {Object} The color table.
 30  */
 31 ColorTable.prototype.addColor = function(options) {
 32 
 33   var requiredOptions = {
 34     name: 'string',
 35     startColor: 'array',
 36     endColor: 'array'
 37   };
 38 
 39   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 40     this[options.name] = {
 41       startColor: options.startColor,
 42       endColor: options.endColor
 43     };
 44   }
 45   return this;
 46 };
 47 
 48 ColorTable.prototype.name = 'ColorTable';
 49 
 50 /**
 51  * Returns start and end colors from a key in the color table.
 52  *
 53  * @param {Object} options A set of options.
 54  *    Required:
 55  *    options.name {number} The name of the entry in the color table.
 56  *    Optional:
 57  *    options.startColor {boolean} Pass true to only return the start color.
 58  *    options.endColor {boolean} Pass true to only return the end color.
 59  * @param {Array} startColor An array representing the start color. ex: [255, 100, 50].
 60  * @param {Array} endColor An array representing the end color. ex: [155, 50, 10].
 61  * @returns {Object|Array} Either an object with startColor and endColor
 62  *    properties or an array representing a start or end color.
 63  *
 64  * @example
 65  * var heat = myColorTable.getColor('heat');
 66  * console.log(heat.startColor); // -> [255, 132, 86]
 67  * console.log(heat.endColor); // -> [175, 47, 0]
 68  *
 69  * var heat = myColorTable.getColor('heat', true);
 70  * console.log(heat); // -> [255, 132, 86]
 71  *
 72  * var heat = myColorTable.getColor('heat', false, true);
 73  * console.log(heat); // -> [175, 47, 0]
 74  */
 75 ColorTable.prototype.getColor = function(name, startColor, endColor) {
 76 
 77   var color, startCol, endCol;
 78 
 79   if (exports.Interface.getDataType(name) === 'string') {
 80 
 81     if (this[name]) {
 82 
 83       color = this[name];
 84 
 85       if (startColor) {
 86         startCol = color.startColor;
 87       }
 88       if (endColor) {
 89         endCol = color.endColor;
 90       }
 91       if (startCol && endCol || !startCol && !endCol) {
 92         return {
 93           startColor: color.startColor,
 94           endColor: color.endColor
 95         };
 96       } else if (startCol) {
 97         return startCol;
 98       } else if (endCol) {
 99         return endCol;
100       }
101     } else {
102       throw new Error('ColorTable: ' + name + ' does not exist. Add colors to the ColorTable via addColor().');
103     }
104   } else {
105     throw new Error('ColorTable: You must pass a name (string) for the color entry in the table.');
106   }
107 };
108 
109 exports.ColorTable = ColorTable;
