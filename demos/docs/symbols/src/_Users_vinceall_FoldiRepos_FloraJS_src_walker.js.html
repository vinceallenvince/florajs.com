  2 /**
  3  * Creates a new Walker.
  4  *
  5  * Walkers have no seeking, steering or directional behavior and just randomly
  6  * explore their World. Use Walkers to create wandering objects or targets
  7  * for Agents to seek. They are not affected by gravity or friction.
  8  *
  9  * @constructor
 10  * @extends Mover
 11  * @param {Object} [opt_options=] A map of initial properties.
 12  */
 13 function Walker(opt_options) {
 14 
 15   var options = opt_options || {};
 16   exports.Mover.call(this, options);
 17   this.name = 'Walker';
 18 }
 19 exports.Utils.extend(Walker, exports.Mover);
 20 
 21 /**
 22  * Initializes an instance.
 23  *
 24  * @param {Object} [opt_options=] A map of initial properties.
 25  * @param {number} [opt_options.width = 10] Width
 26  * @param {number} [opt_options.height = 10] Height
 27  * @param {boolean} [opt_options.isPerlin = true] If set to true, object will use Perlin Noise to calculate its location.
 28  * @param {boolean} [opt_options.remainsOnScreen = false] If set to true and isPerlin = true, object will avoid world edges.
 29  * @param {number} [opt_options.perlinSpeed = 0.005] If isPerlin = true, perlinSpeed determines how fast the object location moves through the noise space.
 30  * @param {number} [opt_options.perlinTime = 0] Sets the Perlin Noise time.
 31  * @param {number} [opt_options.perlinAccelLow = -0.075] The lower bound of acceleration when isPerlin = true.
 32  * @param {number} [opt_options.perlinAccelHigh = 0.075] The upper bound of acceleration when isPerlin = true.
 33  * @param {number} [opt_options.offsetX = Math.random() * 10000] The x offset in the Perlin Noise space.
 34  * @param {number} [opt_options.offsetY = Math.random() * 10000] The y offset in the Perlin Noise space.
 35  * @param {boolean} [opt_options.random = false] Set to true for walker to move in a random direction.
 36  * @param {number} [opt_options.randomRadius = 100] If isRandom = true, walker will look for a new location each frame based on this radius.
 37  * @param {string|Array} [opt_options.color = [255, 150, 50]] Color.
 38  * @param {string|number} [opt_options.borderWidth = '1em'] Border width.
 39  * @param {string} [opt_options.borderStyle = 'double'] Border style.
 40  * @param {string|Array} [opt_options.borderColor = [255, 255, 255]] Border color.
 41  * @param {string} [opt_options.borderRadius = 100] Border radius.
 42  * @param {boolean} [opt_options.avoidWorldEdges = false] If set to true, object steers away from
 43  *    world boundaries.
 44  * @param {number} [opt_options.avoidWorldEdgesStrength = 0] The distance threshold for object
 45  *    start steering away from world boundaries.
 46  */
 47 Walker.prototype.init = function(opt_options) {
 48 
 49   var options = opt_options || {};
 50 
 51   this.width = options.width || 10;
 52   this.height = options.height || 10;
 53   this.perlin = options.perlin || true;
 54   this.remainsOnScreen = !!options.remainsOnScreen;
 55   this.perlinSpeed = options.perlinSpeed || 0.005;
 56   this.perlinTime = options.perlinTime || 0;
 57   this.perlinAccelLow = options.perlinAccelLow || -0.075;
 58   this.perlinAccelHigh = options.perlinAccelHigh || 0.075;
 59   this.offsetX = options.offsetX || Math.random() * 10000;
 60   this.offsetY = options.offsetY || Math.random() * 10000;
 61   this.random = !!options.random;
 62   this.randomRadius = options.randomRadius || 100;
 63   this.color = options.color || [255, 150, 50];
 64   this.borderWidth = options.borderWidth || 2;
 65   this.borderStyle = options.borderStyle || 'solid';
 66   this.borderColor = options.borderColor || [255, 255, 255];
 67   this.borderRadius = options.borderRadius || 100;
 68   this.avoidWorldEdges = !!options.avoidWorldEdges;
 69   this.avoidWorldEdgesStrength = options.avoidWorldEdgesStrength || 50;
 70 };
 71 
 72 /**
 73  */
 74 Walker.prototype.applyForces = function() {
 75 
 76   // walker use either perlin noise or random walk
 77   if (this.perlin) {
 78 
 79     this.perlinTime += this.perlinSpeed;
 80 
 81     if (this.remainsOnScreen) {
 82       this.acceleration = new Burner.Vector();
 83       this.velocity = new Burner.Vector();
 84       this.location.x =  exports.Utils.map(exports.SimplexNoise.noise(this.perlinTime + this.offsetX, 0, 0.1), -1, 1, 0, this.world.bounds[1]);
 85       this.location.y =  exports.Utils.map(exports.SimplexNoise.noise(0, this.perlinTime + this.offsetY, 0.1), -1, 1, 0, this.world.bounds[2]);
 86     } else {
 87       this.acceleration.x =  exports.Utils.map(exports.SimplexNoise.noise(this.perlinTime + this.offsetX, 0, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
 88       this.acceleration.y =  exports.Utils.map(exports.SimplexNoise.noise(0, this.perlinTime + this.offsetY, 0.1), -1, 1, this.perlinAccelLow, this.perlinAccelHigh);
 89     }
 90 
 91   } else if (this.random) {
 92     this.seekTarget = { // find a random point and steer toward it
 93       location: Burner.Vector.VectorAdd(this.location, new Burner.Vector(exports.Utils.getRandomNumber(-this.randomRadius, this.randomRadius), exports.Utils.getRandomNumber(-this.randomRadius, this.randomRadius)))
 94     };
 95     this.applyForce(this._seek(this.seekTarget));
 96   }
 97 
 98   if (this.avoidWorldEdges) {
 99     this._checkAvoidEdges();
100   }
101 };
102 
103 exports.Walker = Walker;
