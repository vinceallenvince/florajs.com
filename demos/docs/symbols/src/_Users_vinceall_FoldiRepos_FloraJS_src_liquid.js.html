1 /*global exports, Burner */
  2 /**
  3  * Creates a new Liquid.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options=] A map of initial properties.
  9  */
 10 function Liquid(opt_options) {
 11   var options = opt_options || {};
 12   options.name = options.name || 'Liquid';
 13   exports.Agent.call(this, options);
 14 }
 15 exports.Utils.extend(Liquid, exports.Agent);
 16 
 17 /**
 18  * Initializes an instance.
 19  *
 20  * @param {Object} [opt_options=] A map of initial properties.
 21  * @param {number} [opt_options.c = 1] Drag coefficient.
 22  * @param {number} [opt_options.mass = 50] Mass.
 23  * @param {boolean} [opt_options.isStatic = true] If true, object will not move.
 24  * @param {number} [opt_options.width = 100] Width.
 25  * @param {number} [opt_options.height = 100] Height.
 26  * @param {number} [opt_options.opacity = 0.75] The particle's opacity.
 27  * @param {number} [opt_options.zIndex = 1] The object's zIndex.
 28  * @param {string|Array} [opt_options.color = [105, 210, 231]] Color.
 29  * @param {string|number} [opt_options.borderWidth = this.width / 4] Border width.
 30  * @param {string} [opt_options.borderStyle = 'double'] Border style.
 31  * @param {string|Array} [opt_options.borderColor = [167, 219, 216]] Border color.
 32  * @param {string} [opt_options.borderRadius = 100] Border radius.
 33  * @param {number} [opt_options.boxShadowSpread = this.width / 4] Box-shadow spread.
 34  * @param {Array} [opt_options.boxShadowColor = [147, 199, 196]] Box-shadow color.
 35  */
 36 Liquid.prototype.init = function(opt_options) {
 37 
 38   var options = opt_options || {};
 39   Liquid._superClass.prototype.init.call(this, options);
 40 
 41   this.c = options.c === 0 ? 0 : options.c || 1;
 42   this.mass = options.mass === 0 ? 0 : options.mass || 50;
 43   this.isStatic = options.isStatic === false ? false : options.isStatic || true;
 44   this.width = options.width === 0 ? 0 : options.width || 100;
 45   this.height = options.height === 0 ? 0 : options.height || 100;
 46   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 47   this.zIndex = options.zIndex === 0 ? 0 : options.zIndex || 1;
 48   this.color = options.color || [105, 210, 231];
 49   this.borderWidth = options.borderWidth || this.width / 4;
 50   this.borderStyle = options.borderStyle || 'double';
 51   this.borderColor = options.borderColor || [167, 219, 216];
 52   this.borderRadius = options.borderRadius || 100;
 53   this.boxShadowSpread = options.boxShadowSpread || this.width / 8;
 54   this.boxShadowColor = options.boxShadowColor || [147, 199, 196];
 55 
 56   Burner.System.updateCache(this);
 57 };
 58 
 59 exports.Liquid = Liquid;
