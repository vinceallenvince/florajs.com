1 /*global exports */
  2 /**
  3  * Creates a new BorderPalette object.
  4  *
  5  * Use this class to create a palette of border styles.
  6  *
  7  * @param {string|number} [opt_id=] An optional id. If an id is not passed, a default id is created.
  8  * @constructor
  9  */
 10 function BorderPalette(opt_id) {
 11 
 12   /**
 13    * Holds a list of border styles.
 14    * @private
 15    */
 16   this._borders = [];
 17 
 18   this.id = opt_id || BorderPalette._idCount;
 19   BorderPalette._idCount++; // increment id
 20 }
 21 
 22 /**
 23  * Increments as each BorderPalette is created.
 24  * @type number
 25  * @default 0
 26  * @private
 27  */
 28 BorderPalette._idCount = 0;
 29 
 30 BorderPalette.prototype.name = 'BorderPalette';
 31 
 32 /**
 33  * Adds a random number of the passed border style to the 'borders' array.
 34  *
 35  * @param {Object} options A set of required options
 36  *    that includes:
 37  *    options.min {number} The minimum number of styles to add.
 38  *    options.max {number} The maximum number of styles to add.
 39  *    options.style {string} The border style.
 40  */
 41 BorderPalette.prototype.addBorder = function(options) {
 42 
 43   var requiredOptions = {
 44     min: 'number',
 45     max: 'number',
 46     style: 'string'
 47   }, i, ln;
 48 
 49   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 50 
 51     ln = exports.Utils.getRandomNumber(options.min, options.max);
 52 
 53     for (i = 0; i < ln; i++) {
 54       this._borders.push(options.style);
 55     }
 56   }
 57   return this;
 58 };
 59 
 60 /**
 61  * @returns A style randomly selected from the 'borders' property.
 62  * @throws {Error} If the 'borders' property is empty.
 63  */
 64 BorderPalette.prototype.getBorder = function() {
 65 
 66   if (this._borders.length > 0) {
 67     return this._borders[exports.Utils.getRandomNumber(0, this._borders.length - 1)];
 68   } else {
 69     throw new Error('BorderPalette.getBorder: You must add borders via addBorder() before using getBorder().');
 70   }
 71 };
 72 exports.BorderPalette = BorderPalette;
