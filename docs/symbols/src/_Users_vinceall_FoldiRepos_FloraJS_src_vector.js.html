  3 /**
  4  * Creates a new Vector.
  5  *
  6  * @param {number} [opt_x = 0] The x location.
  7  * @param {number} [opt_y = 0] The y location.
  8  * @constructor
  9  */
 10 function Vector(opt_x, opt_y) {
 11   var x = opt_x || 0,
 12       y = opt_y || 0;
 13   this.x = x;
 14   this.y = y;
 15 }
 16 
 17 /**
 18  * Subtract two vectors.
 19  *
 20  * @param {number} v1 The first vector.
 21  * @param {number} v2 The second vector.
 22  * @returns {Object} A new Vector.
 23  */
 24 Vector.VectorSub = function(v1, v2) {
 25   return new Vector(v1.x - v2.x, v1.y - v2.y);
 26 };

 28 /**
 29  * Add two vectors.
 30  *
 31  * @param {number} v1 The first vector.
 32  * @param {number} v2 The second vector.
 33  * @returns {Object} A new Vector.
 34  */
 35 Vector.VectorAdd = function(v1, v2) {
 36   return new Vector(v1.x + v2.x, v1.y + v2.y);
 37 };

 61 /**
 62  * Calculates the distance between two vectors.
 63  *
 64  * @param {number} v1 The first vector.
 65  * @param {number} v2 The second vector.
 66  * @returns {number} The distance between the two vectors.
 67  */
 68 Vector.VectorDistance = function(v1, v2) {
 69   return Math.sqrt(Math.pow(v2.x - v1.x, 2) + Math.pow(v2.y - v1.y, 2));
 70 };

 72 /**
 73  * Get the midpoint between two vectors.
 74  *
 75  * @param {number} v1 The first vector.
 76  * @param {number} v2 The second vector.
 77  * @returns {Object} A new Vector.
 78  */
 79 Vector.VectorMidPoint = function(v1, v2) {
 80   return Vector.VectorAdd(v1, v2).div(2); // midpoint = (v1 + v2)/2
 81 };

 83 /**
 84  * Get the angle between two vectors.
 85  *
 86  * @param {number} v1 The first vector.
 87  * @param {number} v2 The second vector.
 88  * @returns {number} An angle.
 89  */
 90 Vector.VectorAngleBetween = function(v1, v2) {
 91   var dot = v1.dot(v2),
 92   theta = Math.acos(dot / (v1.mag() * v2.mag()));
 93   return theta;
 94 };

158 /**
159  * Calculates the magnitude of this vector.
160  *
161  * @returns {number} The vector's magnitude.
162  */
163 Vector.prototype.mag = function() {
164   return Math.sqrt((this.x * this.x) + (this.y * this.y));
165 };
166 
167 /**
168  * Limits the vector's magnitude.
169  *
170  * @param {number} opt_high The upper bound of the vector's magnitude
171  * @param {number} opt_low The lower bound of the vector's magnitude.
172  * @returns {Object} This vector.
173  */
174 Vector.prototype.limit = function(opt_high, opt_low) {
175   var high = opt_high || null,
176       low = opt_low || null;
177   if (high && this.mag() > high) {
178     this.normalize();
179     this.mult(high);
180   }
181   if (low && this.mag() < low) {
182     this.normalize();
183     this.mult(low);
184   }
185   return this;
186 };
187 
188 /**
189  * Divides a vector by its magnitude to reduce its magnitude to 1.
190  * Typically used to retrieve the direction of the vector for later manipulation.
191  *
192  * @returns {Object} This vector.
193  */
194 Vector.prototype.normalize = function() {
195   var m = this.mag();
196   if (m !== 0) {
197     return this.div(m);
198   }
199 };

201 /**
202  * Calculates the distance between this vector and a passed vector.
203  *
204  * @param {Object} vector The target vector.
205  * @returns {Object} The distance between the two vectors.
206  */
207 Vector.prototype.distance = function(vector) {
208   return Math.sqrt(Math.pow(vector.x - this.x, 2) + Math.pow(vector.y - this.y, 2));
209 };

211 /**
212  * Rotates a vector using a passed angle in radians.
213  *
214  * @param {number} radians The angle to rotate in radians.
215  * @returns {Object} This vector.
216  */
217 Vector.prototype.rotate = function(radians) {
218   var cos = Math.cos(radians),
219     sin = Math.sin(radians),
220     x = this.x,
221     y = this.y;
222 
223   this.x = x * cos - y * sin;
224   this.y = x * sin + y * cos;
225   return this;
226 };

228 /**
229  * Calculates the midpoint between this vector and a passed vector.
230  *
231  * @param {Object} v1 The first vector.
232  * @param {Object} v1 The second vector.
233  * @returns {Object} A vector representing the midpoint between the passed vectors.
234  */
235 Vector.prototype.midpoint = function(vector) {
236   return Vector.VectorAdd(this, vector).div(2);
237 };

239 /**
240  * Calulates the dot product.
241  *
242  * @param {Object} vector The target vector.
243  * @returns {Object} A vector.
244  */
245 Vector.prototype.dot = function(vector) {
246   if (this.z && vector.z) {
247     return this.x * vector.x + this.y * vector.y + this.z * vector.z;
248   }
249   return this.x * vector.x + this.y * vector.y;
250 };
