1 /*global exports, Burner, document */
  2 /**
  3  * Creates a new FlowFieldMarker.
  4  *
  5  * @constructor
  6  * @param {Object} options Options.
  7  * @param {Object} options.location Location.
  8  * @param {number} options.scale Scale.
  9  * @param {number} options.opacity Opacity
 10  * @param {number} options.width Width.
 11  * @param {number} options.height Height.
 12  * @param {number} options.angle Angle.
 13  * @param {string} options.colorMode Color mode.
 14  * @param {Object} options.color Color.
 15  */
 16 function FlowFieldMarker(options) {
 17 
 18   var requiredOptions = {
 19         location: 'object',
 20         scale: 'number',
 21         angle: 'number',
 22         opacity: 'number',
 23         width: 'number',
 24         height: 'number',
 25         colorMode: 'string',
 26         color: 'array'
 27       }, el, nose;
 28 
 29   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 30 
 31     el = document.createElement('div');
 32     nose = document.createElement('div');
 33     el.className = 'flowFieldMarker item';
 34     nose.className = 'nose';
 35     el.appendChild(nose);
 36 
 37     el.style.cssText = Burner.System.getCSSText({
 38       x: options.location.x - options.width / 2,
 39       y: options.location.y - options.height / 2,
 40       width: options.width,
 41       height: options.height,
 42       opacity: options.opacity,
 43       angle: options.angle,
 44       colorMode: options.colorMode,
 45       color0: options.color[0],
 46       color1: options.color[1],
 47       color2: options.color[2],
 48       zIndex: options.zIndex,
 49       borderRadius: options.borderRadius
 50     });
 51 
 52     return el;
 53   }
 54 }
 55 
 56 FlowFieldMarker.prototype.name = 'FlowFieldMarker';
 57 
 58 exports.FlowFieldMarker = FlowFieldMarker;
