  2 /**
  3  * Creates a new ColorPalette object.
  4  *
  5  * Use this class to create a palette of colors randomly selected
  6  * from a range created with initial start and end colors. You
  7  * can also generate gradients that smoothly interpolate from
  8  * start and end colors.
  9  *
 10  * @param {string|number} [opt_id=] An optional id. If an id is not passed, a default id is created.
 11  * @constructor
 12  */
 13 function ColorPalette(opt_id) {
 14 
 15   /**
 16    * Holds a list of arrays representing 3-digit color values
 17    * smoothly interpolated between start and end colors.
 18    * @private
 19    */
 20   this._gradients = [];
 21 
 22   /**
 23    * Holds a list of arrays representing 3-digit color values
 24    * randomly selected from start and end colors.
 25    * @private
 26    */
 27   this._colors = [];
 28 
 29   this.id = opt_id || ColorPalette._idCount;
 30   ColorPalette._idCount++; // increment id
 31 }
 32 
 33 /**
 34  * Increments as each ColorPalette is created.
 35  * @type number
 36  * @default 0
 37  * @private
 38  */
 39 ColorPalette._idCount = 0;
 40 
 41 ColorPalette.prototype.name = 'ColorPalette';
 42 
 43 /**
 44  * Creates a color range of 255 colors from the passed start and end colors.
 45  * Adds a random selection of these colors to the color property of
 46  * the color palette.
 47  *
 48  * @param {Object} options A set of required options
 49  *    that includes:
 50  *    options.min {number} The minimum number of colors to add.
 51  *    options.max {number} The maximum number of color to add.
 52  *    options.startColor {Array} The beginning color of the color range.
 53  *    options.endColor {Array} The end color of the color range.
 54  */
 55 ColorPalette.prototype.addColor = function(options) {
 56 
 57   var requiredOptions = {
 58     min: 'number',
 59     max: 'number',
 60     startColor: 'array',
 61     endColor: 'array'
 62   }, i, ln, colors;
 63 
 64   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 65 
 66     ln = exports.Utils.getRandomNumber(options.min, options.max);
 67     colors = ColorPalette._createColorRange(options.startColor, options.endColor, 255);
 68 
 69     for (i = 0; i < ln; i++) {
 70       this._colors.push(colors[exports.Utils.getRandomNumber(0, colors.length - 1)]);
 71     }
 72   }
 73   return this;
 74 };
 75 
 76 /**
 77  * Adds color arrays representing a color range to the gradients property.
 78  *
 79  * @param {Object} options A set of required options
 80  *    that includes:
 81  *    options.startColor {Array} The beginning color of the color range.
 82  *    options.endColor {Array} The end color of the color range.
 83  *    options.totalColors {number} The total number of colors in the gradient.
 84  * @private
 85  */
 86 ColorPalette.prototype.createGradient = function(options) {
 87 
 88   var requiredOptions = {
 89     startColor: 'array',
 90     endColor: 'array'
 91   };
 92 
 93   if (exports.Interface.checkRequiredParams(options, requiredOptions)) {
 94 
 95     this.startColor = options.startColor;
 96     this.endColor = options.endColor;
 97     this.totalColors = options.totalColors || 255;
 98     if (this.totalColors > 0) {
 99       this._gradients.push(ColorPalette._createColorRange(this.startColor, this.endColor, this.totalColors));
100     } else {
101       throw new Error('ColorPalette: total colors must be greater than zero.');
102     }
103   }
104 };
105 
106 /**
107  * @returns An array representing a randomly selected color
108  *    from the colors property.
109  * @throws {Error} If the colors property is empty.
110  */
111 ColorPalette.prototype.getColor = function() {
112 
113   if (this._colors.length > 0) {
114     return this._colors[exports.Utils.getRandomNumber(0, this._colors.length - 1)];
115   } else {
116     throw new Error('ColorPalette.getColor: You must add colors via addColor() before using getColor().');
117   }
118 };
119 
120 /**
121  * Renders a strip of colors representing the color range
122  * in the colors property.
123  *
124  * @param {Object} parent A DOM object to contain the color strip.
125  */
126 ColorPalette.prototype.createSampleStrip = function(parent) {
127 
128   var i, max, div;
129 
130   for (i = 0, max = this._colors.length; i < max; i++) {
131     div = document.createElement('div');
132     div.className = 'color-sample-strip';
133     div.style.background = 'rgb(' + this._colors[i].toString() + ')';
134     parent.appendChild(div);
135   }
136 };

138 /**
139  * Creates an array of RGB color values interpolated between
140  * a passed startColor and endColor.
141  *
142  * @param {Array} startColor The beginning of the color array.
143  * @param {Array} startColor The end of the color array.
144  * @param {number} totalColors The total numnber of colors to create.
145  * @returns {Array} An array of color values.
146  */
147 ColorPalette._createColorRange = function(startColor, endColor, totalColors) {
148 
149   var i, colors = [],
150       startRed = startColor[0],
151       startGreen = startColor[1],
152       startBlue = startColor[2],
153       endRed = endColor[0],
154       endGreen = endColor[1],
155       endBlue = endColor[2],
156       diffRed, diffGreen, diffBlue,
157       newRed, newGreen, newBlue;
158 
159   diffRed = endRed - startRed;
160   diffGreen = endGreen - startGreen;
161   diffBlue = endBlue - startBlue;
162 
163   for (i = 0; i < totalColors; i++) {
164     newRed = parseInt(diffRed * i/totalColors, 10) + startRed;
165     newGreen = parseInt(diffGreen * i/totalColors, 10) + startGreen;
166     newBlue = parseInt(diffBlue * i/totalColors, 10) + startBlue;
167     colors.push([newRed, newGreen, newBlue]);
168   }
169   return colors;
170 };
