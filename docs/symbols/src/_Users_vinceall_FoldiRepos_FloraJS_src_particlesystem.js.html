  2 /**
  3  * Creates a new ParticleSystem.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options=] A map of initial properties.
  9  */
 10 function ParticleSystem(opt_options) {
 11   var options = opt_options || {};
 12   options.name = options.name || 'ParticleSystem';
 13   exports.Agent.call(this, options);
 14 }
 15 exports.Utils.extend(ParticleSystem, exports.Agent);
 16 
 17 /**
 18  * Initializes an instance.
 19  *
 20  * @param {Object} [opt_options=] A map of initial properties.
 21  * @param {boolean} [opt_options.isStatic = true] If set to true, particle system does not move.
 22  * @param {number} [opt_options.lifespan = 1000] The max life of the system. Set to -1 for infinite life.
 23  * @param {number} [opt_options.life = 0] The current life value. If greater than this.lifespan, system is destroyed.
 24  * @param {number} [opt_options.width = 0] Width
 25  * @param {number} [opt_options.height = 0] Height
 26  * @param {number} [opt_options.burst = 1] The number of particles to create per burst.
 27  * @param {number} [opt_options.burstRate = 1] The number of frames between bursts. Lower values = more particles.
 28  * @param {number} [opt_options.emitRadius = 3] The ParticleSystem adds this offset to the location of the Particles it creates.
 29  * @param {Array} [opt_options.startColor = [100, 20, 20]] The starting color of the particle's palette range.
 30  * @param {Array} [opt_options.endColor = [255, 0, 0]] The ending color of the particle's palette range.
 31  * @param {Object} [opt_options.particleOptions] A map of options for particles created by system.
 32  * @param {number} [opt_options.borderWidth = 0] Border width.
 33  * @param {string} [opt_options.borderStyle = 'none'] Border style.
 34  * @param {string|Array} [opt_options.borderColor = 'transparent'] Border color.
 35  * @param {number} [opt_options.borderRadius = 0] Border radius.
 36  */
 37 ParticleSystem.prototype.init = function(opt_options) {
 38 
 39   var options = opt_options || {};
 40   ParticleSystem._superClass.prototype.init.call(this, options);
 41 
 42   this.isStatic = options.isStatic === false ? false : options.isStatic || true;
 43   this.lifespan = options.lifespan === 0 ? 0 : options.lifespan || -1;
 44   this.life = options.life === 0 ? 0 : options.life || 0;
 45   this.width = options.width === 0 ? 0 : options.width || 0;
 46   this.height = options.height === 0 ? 0 : options.height || 0;
 47   this.burst = options.burst === 0 ? 0 : options.burst || 1;
 48   this.burstRate = options.burstRate === 0 ? 0 : options.burstRate || 4;
 49   this.emitRadius = options.emitRadius === 0 ? 0 : options.emitRadius || 3;
 50   this.startColor = options.startColor || [100, 20, 20];
 51   this.endColor = options.endColor || [255, 0, 0];
 52   this.particleOptions = options.particleOptions || {
 53     width : 15,
 54     height : 15,
 55     lifespan : 50,
 56     borderRadius : '100%',
 57     checkEdges : false,
 58     acceleration: null,
 59     velocity: null,
 60     location: null,
 61     maxSpeed: 3,
 62     fade: true
 63   };
 64   this.borderWidth = options.borderWidth || 0;
 65   this.borderStyle = options.borderStyle || 'none';
 66   this.borderColor = options.borderColor || 'transparent';
 67   this.borderRadius = options.borderRadius || 0;
 68   this.clock = 0;
 69 
 70   var pl = new exports.ColorPalette();
 71   pl.addColor({ // adds a random sampling of colors to palette
 72     min: 12,
 73     max: 24,
 74     startColor: this.startColor,
 75     endColor: this.endColor
 76   });
 77 
 78   this.beforeStep = function () {
 79 
 80     var location, offset,
 81         acceleration = this.particleOptions.acceleration,
 82         maxSpeed = this.particleOptions.maxSpeed;
 83 
 84     if (this.life < this.lifespan) {
 85       this.life += 1;
 86     } else if (this.lifespan !== -1) {
 87       Burner.System.destroyItem(this);
 88       return;
 89     }
 90 
 91     if (this.clock % this.burstRate === 0) {
 92 
 93       location = this.getLocation(); // use the system's location
 94       offset = new Burner.Vector(1, 1); // get the emit radius
 95       offset.normalize();
 96       offset.mult(this.emitRadius); // expand emit randius in a random direction
 97       offset.rotate(exports.Utils.getRandomNumber(0, Math.PI * 2, true));
 98       location.add(offset);
 99 
100       for (var i = 0; i < this.burst; i++) {
101         this.particleOptions.world = this.world;
102         this.particleOptions.life = 0;
103         this.particleOptions.color = pl.getColor();
104         this.particleOptions.borderStyle = 'solid';
105         this.particleOptions.borderColor = pl.getColor();
106         this.particleOptions.boxShadowColor = pl.getColor();
107         this.particleOptions.acceleration = new Burner.Vector(1, 1);
108         this.particleOptions.acceleration.normalize();
109         this.particleOptions.acceleration.mult(maxSpeed ? maxSpeed : 3);
110         this.particleOptions.acceleration.rotate(exports.Utils.getRandomNumber(0, Math.PI * 2, true));
111         this.particleOptions.velocity = new Burner.Vector();
112         this.particleOptions.location = ParticleSystem.getParticleLocation(location);
113 
114         Burner.System.add('Particle', this.particleOptions);
115       }
116     }
117     this.clock++;
118   };
119 };

121 /**
122  * Returns a self-executing function that is executed
123  * when particle is initialized. The function retains a
124  * reference to the particle system's location.
125  *
126  * @returns {Function} A function that self-executes and
127  *    returns a reference to the particle system's location.
128  */
129 ParticleSystem.getParticleLocation = function(location) {
130   return (function() {
131     return new Burner.Vector(location.x, location.y);
132   })();
133 };
