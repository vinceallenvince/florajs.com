  2 /**
  3  * Creates a new Cold.
  4  *
  5  * @constructor
  6  * @extends Agent
  7  *
  8  * @param {Object} [opt_options] Options.
  9  * @param {number} [opt_options.mass = 50] Mass.
 10  * @param {boolean} [opt_options.isStatic = true] If true, object will not move.
 11  * @param {number} [opt_options.width = 100] Width.
 12  * @param {number} [opt_options.height = 100] Height.
 13  * @param {number} [opt_options.opacity = 0.75] The opacity.
 14  * @param {number} [opt_options.zIndex = 10] The object's zIndex.
 15  * @param {string|Array} [opt_options.color = [132, 192, 201]] Color.
 16  * @param {string|number} [opt_options.borderWidth = '1em'] Border width.
 17  * @param {string} [opt_options.borderStyle = 'double'] Border style.
 18  * @param {string|Array} [opt_options.borderColor = [0, 89, 102]] Border color.
 19  * @param {string|number} [opt_options.borderRadius = '100%'] Border radius.
 20  */
 21 function Cold(opt_options) {
 22 
 23   var options = opt_options || {};
 24 
 25   exports.Agent.call(this, options);
 26 
 27   this.mass = options.mass === 0 ? 0 : options.mass || 50;
 28   this.isStatic = options.isStatic === false ? false : options.isStatic || true;
 29   this.width = options.width === 0 ? 0 : options.width || 50;
 30   this.height = options.height === 0 ? 0 : options.height || 50;
 31   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 32   this.zIndex = options.zIndex === 0 ? 0 : options.zIndex || 10;
 33   this.color = options.color || [132, 192, 201];
 34   this.borderWidth = options.borderWidth || '1em';
 35   this.borderStyle = options.borderStyle || 'double';
 36   this.borderColor = options.borderColor || [0, 89, 102];
 37   this.borderRadius = options.borderRadius || '100%';
 38 
 39   exports.Burner.PubSub.publish('UpdateCache', this);
 40 }
