  9 /**
 10  * Use to extend the properties and methods of a superClass
 11  * onto a subClass.
 12  */
 13 Utils.extend = function(subClass, superClass) {
 14   function F() {}
 15   F.prototype = superClass.prototype;
 16   subClass.prototype = new F;
 17   subClass.prototype.constructor = subClass;
 18   subClass._superClass = superClass;
 19 };

 21 /**
 22  * Re-maps a number from one range to another.
 23  *
 24  * @param {number} value The value to be converted.
 25  * @param {number} min1 Lower bound of the value's current range.
 26  * @param {number} max1 Upper bound of the value's current range.
 27  * @param {number} min2 Lower bound of the value's target range.
 28  * @param {number} max2 Upper bound of the value's target range.
 29  * @returns {number} A number.
 30  */
 31 Utils.map = function(value, min1, max1, min2, max2) { // returns a new value relative to a new range
 32   var unitratio = (value - min1) / (max1 - min1);
 33   return (unitratio * (max2 - min2)) + min2;
 34 };

 36 /**
 37  * Generates a psuedo-random number within a range.
 38  *
 39  * @param {number} low The low end of the range.
 40  * @param {number} high The high end of the range.
 41  * @param {boolean} [flt] Set to true to return a float.
 42  * @returns {number} A number.
 43  */
 44 Utils.getRandomNumber = function(low, high, flt) {
 45   if (flt) {
 46     return Math.random()*(high-(low-1)) + low;
 47   }
 48   return Math.floor(Math.random()*(high-(low-1))) + low;
 49 };

 51 /**
 52  * Converts degrees to radians.
 53  *
 54  * @param {number} degrees The degrees value to be converted.
 55  * @returns {number} A number in radians.
 56  */
 57 Utils.degreesToRadians = function(degrees) {
 58   if (typeof degrees !== 'undefined') {
 59     return 2 * Math.PI * (degrees/360);
 60   } else {
 61     if (typeof console !== 'undefined') {
 62       console.log('Error: Utils.degreesToRadians is missing degrees param.');
 63     }
 64     return false;
 65   }
 66 };

 85 /**
 86  * Constrain a value within a range.
 87  *
 88  * @param {number} val The value to constrain.
 89  * @param {number} low The lower bound of the range.
 90  * @param {number} high The upper bound of the range.
 91  * @returns {number} A number.
 92  */
 93 Utils.constrain = function(val, low, high) {
 94   if (val > high) {
 95     return high;
 96   } else if (val < low) {
 97     return low;
 98   }
 99   return val;
100 };

115 /**
116  * Add an event listener to a DOM element.
117  *
118  * @param {Object} target The element to receive the event listener.
119  * @param {string} eventType The event type.
120  * @param {function} The function to run when the event is triggered.
121  */
122 Utils.addEvent = function(target, eventType, handler) {
123   if (target.addEventListener) { // W3C
124     this.addEventHandler = function(target, eventType, handler) {
125       target.addEventListener(eventType, handler, false);
126     };
127   } else if (target.attachEvent) { // IE
128     this.addEventHandler = function(target, eventType, handler) {
129       target.attachEvent("on" + eventType, handler);
130     };
131   }
132   this.addEventHandler(target, eventType, handler);
133 };

151 /**
152  * @returns {Object} The current window width and height.
153  * @example getWindowDim() returns {width: 1024, height: 768}
154  */
155 Utils.getWindowSize = function() {
156   var d = {
157     'width' : false,
158     'height' : false
159   };
160   if (typeof(window.innerWidth) !== "undefined") {
161     d.width = window.innerWidth;
162   } else if (typeof(document.documentElement) !== "undefined" &&
163       typeof(document.documentElement.clientWidth) !== "undefined") {
164     d.width = document.documentElement.clientWidth;
165   } else if (typeof(document.body) !== "undefined") {
166     d.width = document.body.clientWidth;
167   }
168   if (typeof(window.innerHeight) !== "undefined") {
169     d.height = window.innerHeight;
170   } else if (typeof(document.documentElement) !== "undefined" &&
171       typeof(document.documentElement.clientHeight) !== "undefined") {
172     d.height = document.documentElement.clientHeight;
173   } else if (typeof(document.body) !== "undefined") {
174     d.height = document.body.clientHeight;
175   }
176   return d;
177 };

179 /**
180  * Returns the data type of the passed argument.
181  *
182  * @param {*} element The variable to test.
183 */
184 Utils.getDataType = function(element) {
185   if (Object.prototype.toString.call(element) === '[object Array]') {
186     return 'array';
187   }
188 
189   if (Object.prototype.toString.call(element) === '[object NodeList]') {
190     return 'nodeList';
191   }
192 
193   return typeof element;
194 };

206 /**
207  * Determines if this object is inside another.
208  *
209  * @param {Object} container The containing object.
210  * @returns {boolean} Returns true if the object is inside the container.
211  */
212 Utils.isInside = function(obj, container) {
213   if (container) {
214     if (obj.location.x + obj.width/2 > container.location.x - container.width/2 &&
215       obj.location.x - obj.width/2 < container.location.x + container.width/2 &&
216       obj.location.y + obj.height/2 > container.location.y - container.height/2 &&
217       obj.location.y - obj.height/2 < container.location.y + container.height/2) {
218       return true;
219     }
220   }
221   return false;
222 };

224 /**
225  * Checks if mouse location is inside the world.
226  *
227  * @param {Object} world A Flora world.
228  * @returns {boolean} True if mouse is inside world; False if
229  *    mouse is outside world.
230  */
231 Utils.mouseIsInsideWorld = function(world) {
232   var mouse = Burner.System.mouse,
233       x = mouse.location.x,
234       y = mouse.location.y,
235       left = world.el.offsetLeft,
236       top = world.el.offsetTop;
237 
238   if (world) {
239     if (x > left &&
240       x < left + world.bounds[1] &&
241       y > top &&
242       y < top + world.bounds[2]) {
243       return true;
244     }
245   }
246   return false;
247 };
