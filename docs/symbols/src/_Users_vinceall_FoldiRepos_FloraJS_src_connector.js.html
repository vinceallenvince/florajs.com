1 /*global exports, Burner */
  2 /**
  3  * Creates a new Connector.
  4  *
  5  * @constructor
  6  * @extends Burner.Item
  7  * @param {Object} [opt_options=] A map of initial properties.
  8  */
  9 function Connector(opt_options) {
 10   var options = opt_options || {};
 11   options.name = options.name || 'Connector';
 12   Burner.Item.call(this, options);
 13 }
 14 exports.Utils.extend(Connector, Burner.Item);
 15 
 16 /**
 17  * Initializes an instance.
 18  *
 19  * @param {Object} options A map of initial properties.
 20  * @param {Object} parentA The object that starts the connection.
 21  * @param {Object} parentB The object that ends the connection.
 22  * @param {number} [options.opacity = 1] Opacity.
 23  * @param {number} [options.zIndex = 0] zIndex.
 24  * @param {number} [options.borderWidth = 1] Border width.
 25  * @param {string} [options.borderStyle = 'dotted'] Border style.
 26  * @param {Array} [options.borderColor = [100, 100, 100]] Border color.
 27  */
 28 Connector.prototype.init = function(options) {
 29 
 30   if (!options || !options.parentA || !options.parentB) {
 31     throw new Error('Connector: both parentA and parentB are required.');
 32   }
 33   this.parentA = options.parentA;
 34   this.parentB = options.parentB;
 35 
 36   this.opacity = options.opacity === 0 ? 0 : options.opacity || 1;
 37   this.zIndex = options.zIndex || 0;
 38 
 39   this.borderWidth = 1;
 40   this.borderRadius = 0;
 41   this.borderStyle = 'dotted';
 42   this.borderColor = [150, 150, 150];
 43 
 44   this.width = 0;
 45   this.height = 0;
 46   this.color = 'transparent';
 47 };
 48 
 49 Connector.prototype.name = 'Connector';
 50 
 51 /**
 52  * Called every frame, step() updates the instance's properties.
 53  */
 54 Connector.prototype.step = function() {
 55 
 56   var a = this.parentA.location,
 57       b = this.parentB.location;
 58 
 59   this.width = Math.floor(Burner.Vector.VectorSub(this.parentA.location,
 60       this.parentB.location).mag());
 61 
 62   this.location = Burner.Vector.VectorAdd(this.parentA.location,
 63       this.parentB.location).div(2); // midpoint = (v1 + v2)/2
 64 
 65   this.angle = exports.Utils.radiansToDegrees(Math.atan2(b.y - a.y, b.x - a.x) );
 66 };
 67 
 68 exports.Connector = Connector;
