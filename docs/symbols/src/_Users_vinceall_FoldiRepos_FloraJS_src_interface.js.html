1 /*global exports, console */
  2 /** @namespace */
  3 var Interface = {};
  4 
  5 /**
  6  * @description Compares passed options to a set of required options.
  7  * @param {Object} optionsPassed An object containing options passed to a function.
  8  * @param {Object} optionsRequired An object containing a function's required options.
  9  * @param {string=} opt_from A message or identifier from the calling function.
 10  * @returns {boolean} Returns true if all required options are present and of the right data type.
 11  *     Returns false if any required options are missing or are the wrong data type. Also returns false
 12  *     if any of the passed options are empty.
 13  * @example
 14  * optionsPassed = {
 15  *  form: document.getElementById("form"),
 16  *    color: "blue",
 17  *  total: "a whole bunch",
 18  *   names: ["jimmy", "joey"]
 19  * }
 20  * optionsRequired = {
 21  *   form: "object",
 22  *   color: "string",
 23  *   total: "number|string",
 24  *   names: "array"
 25  * }
 26  * checkRequiredOptions(optionsPassed, optionsRequired) returns true in this case.
 27  *
 28  * Notice you can separate required options data types with a pipe character to allow multiple data types.
 29  */
 30 Interface.checkRequiredParams = function(optionsPassed, optionsRequired, opt_from) {
 31 
 32   var i, msg, check = true;
 33 
 34   for (i in optionsRequired) { // loop thru required options
 35 
 36     if (optionsRequired.hasOwnProperty(i)) {
 37 
 38       try {
 39 
 40         // if no options were passed
 41         if (typeof optionsPassed === "undefined") {
 42           throw new Error('checkRequiredOptions: No options were passed.');
 43         }
 44 
 45         // if there is not a corresponding key in the passed options; or options passed value is blank
 46         if (!this.checkDataType(this.getDataType(optionsPassed[i]), optionsRequired[i].split('|')) || optionsPassed[i] === "") {
 47 
 48           check = false;
 49 
 50           if (optionsPassed[i] === '') {
 51             msg = 'checkRequiredOptions: required option "' + i + '" is empty.';
 52           } else if (typeof optionsPassed[i] === 'undefined') {
 53             msg = 'checkRequiredOptions: required option "' + i + '" is missing from passed options.';
 54           } else {
 55             msg = 'checkRequiredOptions: passed option "' + i + '" must be type ' + optionsRequired[i] +
 56             '. Passed as ' + this.getDataType(optionsPassed[i]) + '.';
 57           }
 58 
 59           throw new Error(msg);
 60         }
 61       } catch (err) {
 62         if (typeof console !== 'undefined') {
 63           console.log('ERROR: ' + err.message + (opt_from ? ' from: ' + opt_from : ''));
 64         }
 65       }
 66     }
 67   }
 68   return check;
 69 };
 70 
 71 /**
 72  * Loops through an array of data types and checks if the
 73  * passed option matches any entry.
 74  *
 75  * @param {string} option The data type to check.
 76  * @param {array} typesToMatch An array of data types to check option against.
 77  * @returns {boolean} If option matches any entry in typesToMatch,
 78  * return true. Else, returns false.
 79  */
 80 Interface.checkDataType = function(option, typesToMatch) {
 81 
 82   var i, max;
 83 
 84   for (i = 0, max = typesToMatch.length; i < max; i++) {
 85     if (option === typesToMatch[i]) {
 86       return true;
 87     }
 88   }
 89   return false;
 90 };
 91 
 92 /**
 93  * Checks for data type.
 94  *
 95  * @returns {string} The data type of the passed variable.
 96  */
 97 Interface.getDataType = function(element) {
 98 
 99   if (Object.prototype.toString.call(element) === '[object Array]') {
100     return 'array';
101   }
102 
103   if (Object.prototype.toString.call(element) === '[object NodeList]') {
104     return 'nodeList';
105   }
106 
107   return typeof element;
108 };
