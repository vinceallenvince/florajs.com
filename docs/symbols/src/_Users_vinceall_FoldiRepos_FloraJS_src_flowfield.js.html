1 /*global exports, Burner */
  2 /**
  3  * Creates a new FlowField.
  4  *
  5  * @constructor
  6  */
  7 function FlowField(opt_options) {
  8   var options = opt_options || {};
  9   options.name = options.name || 'FlowField';
 10   Burner.Item.call(this, options);
 11 }
 12 exports.Utils.extend(FlowField, Burner.Item);
 13 
 14 /**
 15  * Initializes an instance.
 16  *
 17  * @param {Object} [opt_options=] A map of initial properties.
 18  * @param {number} [opt_options.resolution = 50] The lower the value, the more vectors are created
 19  *    to define the flow field. Low values increase processing time to create the field.
 20  * @param {number} [opt_options.perlinSpeed = 0.01] The speed to move through the Perlin Noise space.
 21  * @param {number} [opt_options.perlinTime = 100] Sets the Perlin Noise time.
 22  * @param {Object} [opt_options.field = null] A list of vectors that define the flow field.
 23  * @param {Object} [opt_options.createMarkers = false] Set to true to visualize the flow field.
 24  */
 25 FlowField.prototype.init = function(opt_options) {
 26 
 27   var options = opt_options || {};
 28 
 29   this.resolution = options.resolution || 50;
 30   this.perlinSpeed = options.perlinSpeed || 0.01;
 31   this.perlinTime = options.perlinTime || 100;
 32   this.field = options.field || null;
 33   this.createMarkers = options.createMarkers || false;
 34   // if a world is not passed, use the first world in the system
 35   this.world = options.world || Burner.System.firstWorld();
 36 };
 37 
 38 FlowField.prototype.name = 'FlowField';
 39 
 40 /**
 41  * Builds a FlowField.
 42  */
 43 FlowField.prototype.build = function() {
 44 
 45   var col, colMax, row, rowMax, x, y, theta, fieldX, fieldY, field, angle,
 46       vectorList = {},
 47       world = this.world,
 48       cols = Math.ceil(world.bounds[1] / parseFloat(this.resolution)),
 49       rows = Math.ceil(world.bounds[2] / parseFloat(this.resolution)),
 50       xoff = this.perlinTime, // create markers and vectors
 51       yoff;
 52 
 53   for (col = 0, colMax = cols; col < colMax ; col += 1) {
 54     yoff = this.perlinTime;
 55     vectorList[col] = {};
 56     for (row = 0, rowMax = rows; row < rowMax ; row += 1) {
 57 
 58       x = col * this.resolution + this.resolution / 2; // get location on the grid
 59       y = row * this.resolution + this.resolution / 2;
 60 
 61       theta = exports.Utils.map(exports.SimplexNoise.noise(xoff, yoff, 0.1), 0, 1, 0, Math.PI * 2); // get the vector based on Perlin noise
 62       fieldX = Math.cos(theta);
 63       fieldY = Math.sin(theta);
 64       field = new Burner.Vector(fieldX, fieldY);
 65       vectorList[col][row] = field;
 66       angle = exports.Utils.radiansToDegrees(Math.atan2(fieldY, fieldX)); // get the angle of the vector
 67 
 68       if (this.createMarkers) {
 69 
 70         var ffm = new exports.FlowFieldMarker({ // create the marker
 71           location: new Burner.Vector(x, y),
 72           scale: 1,
 73           opacity: exports.Utils.map(angle, -360, 360, 0.1, 0.75),
 74           width: this.resolution,
 75           height: this.resolution/2,
 76           field: field,
 77           angle: angle,
 78           colorMode: 'rgb',
 79           color: [200, 100, 50],
 80           borderRadius: 0,
 81           zIndex: 0
 82         });
 83         world.el.appendChild(ffm);
 84       }
 85       yoff += parseFloat(this.perlinSpeed);
 86     }
 87     xoff += parseFloat(this.perlinSpeed);
 88   }
 89   this.field = vectorList;
 90 };
 91 
 92 exports.FlowField = FlowField;
