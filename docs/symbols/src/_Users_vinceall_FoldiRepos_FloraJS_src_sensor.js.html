  2 /**
  3  * Creates a new Sensor object.
  4  *
  5  * @constructor
  6  * @extends Mover
  7  *
  8  * @param {Object} [opt_options=] A map of initial properties.
  9  */
 10 function Sensor(opt_options) {
 11   var options = opt_options || {};
 12   options.name = options.name || 'Sensor';
 13   exports.Mover.call(this, options);
 14 }
 15 exports.Utils.extend(Sensor, exports.Mover);

 17 /**
 18  * Initializes an instance.
 19  *
 20  * @param {Object} [opt_options=] A map of initial properties.
 21  * @param {string} [opt_options.type = ''] The type of stimulator that can activate this sensor. eg. 'cold', 'heat', 'light', 'oxygen', 'food', 'predator'
 22  * @param {string} [opt_options.behavior = 'LOVE'] The vehicle carrying the sensor will invoke this behavior when the sensor is activated.
 23  * @param {number} [opt_options.sensitivity = 2] The higher the sensitivity, the farther away the sensor will activate when approaching a stimulus.
 24  * @param {number} [opt_options.width = 5] Width.
 25  * @param {number} [opt_options.height = 5] Height.
 26  * @param {number} [opt_options.offsetDistance = 30] The distance from the center of the sensor's parent.
 27  * @param {number} [opt_options.offsetAngle = 0] The angle of rotation around the vehicle carrying the sensor.
 28  * @param {number} [opt_options.opacity = 0.75] Opacity.
 29  * @param {Object} [opt_options.target = null] A stimulator.
 30  * @param {boolean} [opt_options.activated = false] True if sensor is close enough to detect a stimulator.
 31  * @param {Array} [opt_options.activatedColor = [255, 255, 255]] The color the sensor will display when activated.
 32  * @param {number} [opt_options.borderRadius = 100] Border radius.
 33  * @param {number} [opt_options.borderWidth = 2] Border width.
 34  * @param {string} [opt_options.borderStyle = 'solid'] Border style.
 35  * @param {Array} [opt_options.borderColor = [255, 255, 255]] Border color.
 36  */
 37 Sensor.prototype.init = function(opt_options) {
 38 
 39   var options = opt_options || {};
 40   Sensor._superClass.prototype.init.call(this, options);
 41 
 42   this.type = options.type || '';
 43   this.behavior = options.behavior || 'LOVE';
 44   this.sensitivity = options.sensitivity === 0 ? 0 : options.sensitivity || 2;
 45   this.width = options.width === 0 ? 0 : options.width || 7;
 46   this.height = options.height === 0 ? 0 : options.height || 7;
 47   this.offsetDistance = options.offsetDistance === 0 ? 0 : options.offsetDistance|| 30;
 48   this.offsetAngle = options.offsetAngle || 0;
 49   this.opacity = options.opacity === 0 ? 0 : options.opacity || 0.75;
 50   this.target = options.target || null;
 51   this.activated = !!options.activated;
 52   this.activatedColor = options.activatedColor || [255, 255, 255];
 53   this.borderRadius = 100;
 54   this.borderWidth = 2;
 55   this.borderStyle = 'solid';
 56   this.borderColor = [255, 255, 255];
 57 };

 61 /**
 62  * Called every frame, step() updates the instance's properties.
 63  */
 64 Sensor.prototype.step = function() {
 65 
 66   var check = false, i, max;
 67 
 68   var heat = Burner.System._caches.Heat || {list: []},
 69       cold = Burner.System._caches.Cold || {list: []},
 70       predators = Burner.System._caches.Predators || {list: []},
 71       lights = Burner.System._caches.Light || {list: []},
 72       oxygen = Burner.System._caches.Oxygen || {list: []},
 73       food = Burner.System._caches.Food || {list: []};
 74 
 75   // what if cache does not exist?
 76 
 77   if (this.type === 'heat' && heat.list && heat.list.length > 0) {
 78     for (i = 0, max = heat.list.length; i < max; i++) { // heat
 79       if (this.isInside(this, heat.list[i], this.sensitivity)) {
 80         this.target = heat.list[i]; // target this stimulator
 81         this.activated = true; // set activation
 82         check = true;
 83       }
 84     }
 85   } else if (this.type === 'cold' && cold.list && cold.list.length > 0) {
 86     for (i = 0, max = cold.list.length; i < max; i++) { // cold
 87       if (this.isInside(this, cold.list[i], this.sensitivity)) {
 88         this.target = cold.list[i]; // target this stimulator
 89         this.activated = true; // set activation
 90         check = true;
 91       }
 92     }
 93   } else if (this.type === 'predator' && predators.list && predators.list.length > 0) {
 94     for (i = 0, max = predators.list.length; i < max; i += 1) { // predator
 95       if (this.isInside(this, predators.list[i], this.sensitivity)) {
 96         this.target = predators.list[i]; // target this stimulator
 97         this.activated = true; // set activation
 98         check = true;
 99       }
100     }
101   } else if (this.type === 'light' && lights.list && lights.list.length > 0) {
102     for (i = 0, max = lights.list.length; i < max; i++) { // light
103       // check the obj has not been marked as deleted
104       if (lights.lookup[lights.list[i].id]) {
105         if (this.isInside(this, lights.list[i], this.sensitivity)) {
106           this.target = lights.list[i]; // target this stimulator
107           this.activated = true; // set activation
108           check = true;
109         }
110       }
111     }
112   } else if (this.type === 'oxygen' && oxygen.list && oxygen.list.length > 0) {
113     for (i = 0, max = oxygen.list.length; i < max; i += 1) { // oxygen
114       // check the obj has not been marked as deleted
115       if (oxygen.lookup[oxygen.list[i].id]) {
116         if (this.isInside(this, oxygen.list[i], this.sensitivity)) {
117           this.target = oxygen.list[i]; // target this stimulator
118           this.activated = true; // set activation
119           check = true;
120         }
121       }
122     }
123   } else if (this.type === 'food' && food.list && food.list.length > 0) {
124     for (i = 0, max = food.list.length; i < max; i += 1) { // food
125       // check the obj has not been marked as deleted
126       if (food.lookup[food.list[i].id]) {
127         if (this.isInside(this, food.list[i], this.sensitivity)) {
128           this.target = food.list[i]; // target this stimulator
129           this.activated = true; // set activation
130           check = true;
131         }
132       }
133     }
134   }
135   if (!check) {
136     this.target = null;
137     this.activated = false;
138     this.color = 'transparent';
139   } else {
140     this.color = this.activatedColor;
141   }
142   if (this.afterStep) {
143     this.afterStep.apply(this);
144   }
145 
146 };

148 /**
149  * Returns a force to apply to an agentwhen its sensor is activated.
150  *
151  */
152 Sensor.prototype.getActivationForce = function(agent) {
153 
154   var distanceToTarget, m, v, steer;
155 
156   switch (this.behavior) {
157 
158     /**
159      * Steers toward target
160      */
161     case "AGGRESSIVE":
162       return this._seek(this.target);
163 
164     /**
165      * Steers away from the target
166      */
167     case "COWARD":
168       var f = this._seek(this.target);
169       return f.mult(-1);
170 
171     /**
172      * Speeds toward target and keeps moving
173      */
174     case "LIKES":
175       var dvLikes = Burner.Vector.VectorSub(this.target.location, this.location);
176       distanceToTarget = dvLikes.mag();
177       dvLikes.normalize();
178 
179       m = distanceToTarget/agent.maxSpeed;
180       dvLikes.mult(m);
181 
182       steer = Burner.Vector.VectorSub(dvLikes, agent.velocity);
183       steer.limit(agent.maxSteeringForce);
184       return steer;
185 
186     /**
187      * Arrives at target and remains
188      */
189     case "LOVES":
190       var dvLoves = Burner.Vector.VectorSub(this.target.location, this.location); // desiredVelocity
191       distanceToTarget = dvLoves.mag();
192       dvLoves.normalize();
193 
194       if (distanceToTarget > this.width) {
195         m = distanceToTarget/agent.maxSpeed;
196         dvLoves.mult(m);
197         steer = Burner.Vector.VectorSub(dvLoves, agent.velocity);
198         steer.limit(agent.maxSteeringForce);
199         return steer;
200       }
201       agent.velocity = new Burner.Vector();
202       agent.acceleration = new Burner.Vector();
203       return new Burner.Vector();
204 
205     /**
206      * Arrives at target but does not stop
207      */
208     case "EXPLORER":
209 
210       var dvExplorer = Burner.Vector.VectorSub(this.target.location, this.location);
211       distanceToTarget = dvExplorer.mag();
212       dvExplorer.normalize();
213 
214       m = distanceToTarget/agent.maxSpeed;
215       dvExplorer.mult(-m);
216 
217       steer = Burner.Vector.VectorSub(dvExplorer, agent.velocity);
218       steer.limit(agent.maxSteeringForce * 0.05);
219       return steer;
220 
221     /**
222      * Moves in the opposite direction as fast as possible
223      */
224     /*case "RUN":
225       return this.flee(this.target);*/
226 
227     case "ACCELERATE":
228       v = agent.velocity.clone();
229       v.normalize();
230       return v.mult(agent.minSpeed);
231 
232     case "DECELERATE":
233       v = agent.velocity.clone();
234       v.normalize();
235       return v.mult(-agent.minSpeed);
236 
237     default:
238       return new Burner.Vector();
239   }

242 /**
243  * Checks if a sensor can detect a stimulator.
244  *
245  * @param {Object} params The sensor.
246  * @param {Object} container The stimulator.
247  * @param {number} sensitivity The sensor's sensitivity.
248  */
249 Sensor.prototype.isInside = function(item, container, sensitivity) {
250 
251   if (item.location.x + item.width/2 > container.location.x - container.width/2 - (sensitivity * container.width) &&
252     item.location.x - item.width/2 < container.location.x + container.width/2 + (sensitivity * container.width) &&
253     item.location.y + item.height/2 > container.location.y - container.height/2 - (sensitivity * container.height) &&
254     item.location.y - item.height/2 < container.location.y + container.height/2 + (sensitivity * container.height)) {
255     return true;
256   }
257   return false;
258 };
